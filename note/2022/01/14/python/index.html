linalg.solve

x

Matplotlib

Draw a plot:(Template)

t = np.arange(start=0., stop=5., step=0.2)
plt.plot(t, t, 'r--')
plt.plot(t, t**2, 'bs')
plt.plot(t, t**3, 'g^')
plt.xlabel('$t$')
plt.ylabel('$f(t)$')
plt.title('Curves')
plt.legend(['$t$', '$t^2$', '$t^3$'])
plt.show()

Subplot

def f(t):
    return np.exp(-t) * np.cos(2*np.pi*t)

t1 = np.arange(0.0, 5.0, 0.1)
t2 = np.arange(0.0, 5.0, 0.02)

plt.figure(1)
plt.subplot(211)
plt.plot(t1, f(t1), 'bo', t2, f(t2), 'k')

plt.subplot(212)
plt.plot(t2, np.cos(2*np.pi*t2), 'r--')
plt.show()
